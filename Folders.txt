├── _data/
│   ├── navigation.yml              # منوی اصلی تو در تو (Mega Menu)
│   └── board_structure.yml         # ساختار رتبه‌بندی اعضای هیئت مدیره
├── _layouts/                       # قالب‌های پایه و ساختارهای تو در تو
│   ├── default.html                # شاسی اصلی سایت (Header/Footer/Scripts)
│   ├── page.html                   # صفحات ایستا (درباره ما، قوانین)
│   ├── member_profile.html         # قالب نهایی پروفایل اختصاصی هر عضو/ادیتور
│   ├── journal_structure.html      # قالب صفحات تو در تو مجلات و ساختار ادیتوریال
│   └── event_detail.html           # قالب جامع صفحات کنفرانس و کارگاه‌ها
├── _includes/                      # تکه‌های کد تکرارشونده
│   ├── header.html                 # هدر و منوهای ناوبری
│   ├── footer.html                 # فوتر و فرم خبرنامه
│   ├── member_card.html            # کامپوننت کارت معرفی اعضا در صفحات عمومی
│   └── gallery.html                # ماژول گالری تصاویر پیشرفته
├── _editorial_board/               # کالکشن تو در تو: اعضای هیئت تحریریه مجلات
│   ├── journal-a/                  # مجله اول (مثلاً Journal of Decision Analytics)
│   │   ├── editor-in-chief.md
│   │   ├── associate-editor-1.md
│   │   └── reviewer-1.md

│       ├── editor-in-chief.md
│       └── editorial-member-1.md
├── _members/                       # کالکشن تو در تو: اعضای عمومی انجمن و پورتال‌ها
│   ├── board-members/              # اعضای هیئت مدیره انجمن
│   │   └── member-01.md
│   ├── fellows/                    # اعضای برجسته (Fellows)
│   │   └── fellow-01.md
│   └── regular-members/            # اعضای عادی و دانشجویی
│       └── student-01.md
├── _events/                        # کالکشن تو در تو: کنفرانس‌ها، وبینارها و رویدادها
│   ├── conferences/                # کنفرانس‌های سالانه
│   │   ├── cie-conference-2025.md
│   │   └── annual-conference-2026.md
│   └── workshops/                  # کارگاه‌های آموزشی و سمینارها
│       └── mcdm-workshop-2026.md
├── assets/                         # دارایی‌های ایستا (Static Assets)
│   ├── css/
│   │   └── main.css                # استایل‌های سراسری، حالت تاریک/روشن و گریدها
│   ├── js/
│   │   └── automation.js           # اسکریپت‌های اتوماسیون فرم‌ها و تعاملات
│   └── images/                     # پوشه جامع تصاویر با تفکیک دقیق
│       ├── avatars/                # عکس پرسنلی اعضا و ادیتورها
│       ├── journals/               # کاور مجلات و لوگوها
│       └── event-galleries/        # پوشه‌های مجزا برای عکس‌های هر رویداد
└── index.md                        # صفحه اصلی سایت (شامل بخش‌های داینامیک)
